-- Create tables
CREATE TABLE IF NOT EXISTS active_lounges (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  lounge_name text NOT NULL,
  host_name text NOT NULL,
  host_id text NOT NULL,
  entry_code text NOT NULL,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS class_sessions (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  lounge_id uuid REFERENCES active_lounges(id) ON DELETE CASCADE,
  status text DEFAULT 'not_started',
  current_bookmark int DEFAULT 0,
  started_at timestamptz,
  last_updated timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS questions (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  lounge_id uuid REFERENCES active_lounges(id) ON DELETE CASCADE,
  user_name text NOT NULL,
  question_text text NOT NULL,
  bookmark_number int DEFAULT 0,
  answered boolean DEFAULT false,
  peer_id text,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS bookmarks (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  lounge_id uuid REFERENCES active_lounges(id) ON DELETE CASCADE,
  bookmark_number int NOT NULL,
  timestamp bigint,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS file_shares (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  lounge_id uuid REFERENCES active_lounges(id) ON DELETE CASCADE,
  from_peer_id text,
  from_name text,
  to_peer_id text,
  file_name text NOT NULL,
  file_url text NOT NULL,
  created_at timestamptz DEFAULT now()
);
